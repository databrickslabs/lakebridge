FETCH   
          [ [ NEXT | PRIOR | FIRST | LAST

               ]   
               FROM   
          ]   
{ { [ GLOBAL ] cursor_name } | @cursor_variable_name }   
[ INTO @variable_name [ ,...n ] ]
